void nhap(int a[],int n)
{
    for(int i=0; i<n; i++)
    {
        scanf("%d",&a[i]);
    }
}
void xuat(int a[],int n)
{
    for(int i=0; i<n; i++)
    {
        printf("%-5d",a[i]);
    }
}
void ghep(int a[],int b[],int c[],int k) //k=m+n 
{
    int i=0,j=0,t;
    for(t=0; t<k; t++)
    {
        c[t]=(a[i]<b[j])?a[i++]:b[j++];
    }
}
void sapxep(int a[],int n)
{
    for(int i =0; i<n-1; i++)
    {
        for(int j =i+1; j<n; j++)
        {
            if(a[i]>a[j])
            {
                int t=a[i];
                a[i]=a[j];
                a[j] = t;
            }
        }

    }
}
